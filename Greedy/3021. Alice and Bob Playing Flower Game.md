#define ll long long
class Solution {
public:
    long long flowerGame(int n, int m) {
        ll c = n / 2;
        ll d = (n / 2) + (n % 2 != 0);
        ll c1 = m / 2;
        ll d1 = (m / 2) + (m % 2 != 0);
        return (c * d1) + (d * c1);
    }
};
