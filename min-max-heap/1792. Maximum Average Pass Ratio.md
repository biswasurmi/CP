class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& cls, int ex) {
        priority_queue<pair<double, pair<double, double>>>pq;
        for(auto i : cls){
            double c = i[0], d = i[1];
            double dif = ((c + 1) / (d + 1)) - (c / d);
            pq.push({dif, {c, d}});
        }
        while(ex--){
            auto it = pq.top();
            double c = it.first, d = it.second.first, e = it.second.second;
            pq.pop();
            d++; e++;
            double dif = ((d + 1) / (e + 1)) - (d / e);
            pq.push({dif, {d, e}});
        }
        double res = 0.0;
        while(!pq.empty()){
            auto it = pq.top();
            pq.pop();
            double c = it.second.first, d = it.second.second;
            res += (c / d);
        }
        return res / (cls.size() * 1.0);
    }
};
