class Solution {
public:
    vector<int> findDiagonalOrder(vector<vector<int>>& mat) {
        vector<int>res;
        for(int i = 0; i < mat[0].size(); i++){
            vector<int>v;
            for(int j = i, k = 0; j >= 0 and k < mat.size(); j--, k++){
                v.push_back(mat[k][j]);
            }
            if(i % 2 == 0) reverse(v.begin(), v.end());
            for(auto j : v) res.push_back(j);
        }
        for(int i1 = 1, i = mat[0].size() - 1; i1 < mat.size(); i1++){
            vector<int>v;
            for(int j = i, k = i1; k < mat.size() and j >= 0; j--, k++){
                v.push_back(mat[k][j]);
            }
            int mn = mat[0].size();
            if((mn % 2 == 0 and i1 % 2) or (mn % 2 and i1 % 2 == 0)) reverse(v.begin(), v.end());
            for(auto j : v) res.push_back(j);
        }
        return res;
    }
};
