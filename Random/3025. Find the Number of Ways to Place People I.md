class Solution {
public:
    int numberOfPairs(vector<vector<int>>& points) {
        int res = 0;
        for(int i = 0; i < points.size(); i++){
            for(int j = 0; j < points.size(); j++){
                if(i == j) continue;
                int c = points[i][0], d = points[i][1], c1 = points[j][0], d1 = points[j][1];
                if(c1 <= c and d1 >= d){
                    bool ok = false;
                    for(int k = 0; k < points.size(); k++){
                        if(i == k or j == k) continue;
                        int c2 = points[k][0], d2 = points[k][1];
                        int cc1 = min(c, c1), cc2 = max(c, c1), dd1 = min(d, d1), dd2 = max(d, d1);
                        if(c2 >= cc1 and c2 <= cc2 and d2 >= dd1 and d2 <= dd2){
                            ok = true;
                            break;
                        }
                    }
                    if(!ok){
                        res++;
                    } 
                }
                
            }
        }
        return res;
    }
};
